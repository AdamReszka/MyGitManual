git init - zainicjowanie projektu lub rozpoczęcie pracy z git w istniejącym lokalnie projekcie

git init projectName - zainicjowanie projektu o wybranej nazwie utworzy folder dla projektu

git status - sprawdzenie statusu zmian, wylistowanie plików w których zostały dokonae zmiany

git add nazwa.pliku Dopisanie zmian z danego pliku do stage'a czyli do przestrzeni z której są zapisywane i wysyłane. Nie musimy zapisywać wszystkiego dlatego umieszczając zmiany w stage'u określamy, że to one będą przeznaczone do zapisu.

git add . lub git add -A - przygotowanie wszystkich zmian do zapisu

git checkout -- . - odrzucenie, usunięcie wszystkich wprowadzonych niezapisanych do tej pory zmian

git checkout -- file.name - odrzucenie, usunięcie wprowadzonych niezapisanych do tej pory zmian ze wskazanego pliku

git commit - lokalne zatwierdzenie zmian z poczekalni (stage)

git commit -v - lokalne zapisanie zmian ze stage'a, dzięki fladze -v do komentarza zostaną dodane wiersze, które zostały zmienione

git commit -m "example message content" - lokalne zapisanie zmian ze stage'a, dzięki fladze message możemy opisać jakie zmiany wprowadziliśmy

git commit -a -m "example message content" - lokalne zapisanie zmian z automatycznym dodaniem zmian do stage'a, nie trzeba wyknywać "git add"

git commit --amend - dogrywa ostatnie zmiany aktualizując ostatni commit (nie tworzy nowego)

git config --global user.name "Jan Kowalski" - lokalne ustawienie nazwy użytkownika konieczne do jego identyfikacji

git config --global user.email "jan.kowalski@example.com" - lokalne ustawienia adresu email użytkownika

git config --list - listuje bieżące ustawienia konfiguracji danego projektu

git clone git://github.com/jankowalski/nowyprojekt.git - klonowanie do własnej lokalizacji istniejącego zdalnie repozytorium

git clone git://github.com/jankowalski/nowyprojekt.git directoryName - klonowanie do własnej lokalizacji istniejącego zdalnie repozytorium plus własna definicja nazwy lokalizacji

git remote - pokazuje nazwy wszystkich aktualnie zdefiniowanych serwerów dla zdalnych repozytoriów (chociaż jeden)

git remote -v - dodatkowo pokazuje zapamiętany pełen adres URL dla każdej lokalizacji

git remote add origin git://github.com/jankowalski/nowyprojekt.git - przypisanie tego zdalnego repozytorium jako głównej ścieżki na której znajdzie się master branch i inne branche. Origin ustawiamy po to by nie powtarzać ciągle całego linku do repo.

git remote set-url origin git://github.com/jankowalski/nowyprojekt.git - zmiana na nowy, adresu zdalnego repozytorium.

git remote show origin - pokaże dodatkowe informacje o wybranym repo (adres URL czy info o śledzonej gałęzi)

git remote rename stara.nazwa nowa.nazwa - pozwala na zmianę odnośnika czyli skrótu przypisanego do repozytorium

git remote rm moja.nazwa - pozwala na zmianę usunięcie odnośnika

git diff - pokaże szczegółowo wszystkie zmiany, które nie trafiły jeszcze do poczekalni (stage)

git diff --staged - pokaże szczegółowo wszystkie zmiany w poczekalni (stage), które trafią do zatwierdzenia

git fetch origin - pobiera zmiany, których nie posiadasz ale w odróżnieniu od pulla nie wykonuje merga, jedynie pobiera zmiany

git pull - pobiera wszystkie najnowsze dane z serwera i dokonuje próby scalenia ich z lokalnym repo, scala je jeśli nie ma konfliktów

git push origin master - wysłanie zapisanych zmian na master branch zdalnego repozytorium

git help "polecenie", git "polecenie" --help, man git-"polecenie" (np. git help config) - uzyskanie pomocy w kwestii danego polecenia

git rm nazwa.pliku - usuwa plik z gita i z katalogu roboczego

git rm -f nazwa.pliku - usuwa plik, który był modyfikowany zmiany w nim nie zostały zapamiętane poprzez commit

git rm --cached nazwa.pliku - usunięcie pliku ze stage'a ale pozostawienie go na dysku. Można dzięki temu dodać potem swobodnie plik do .gitignore

git rm log/\*.log - zostaną usunięte wszystkie pliki o rozszerzeniu .log z katalogu log/

git mv file_from file_to - pomoże w zmianie nazwy pliku

git log - uruchamia podgląd historii zapisywanych zmian, zaczynając od najnowszych commitów

git log -p -3 - pokaże szczegółowy podgląd zmian z ograniczeniem do trzech ostatnich commitów

git log --stat - pokaże podgląd zmian ze skróconymi statystykami commitów

git log --pretty=oneline - wyświetla wyniki w jednej linii, dodatkowo dostępne są jeszcze opcje "short", "full", "fuller", zawierają one więcej informacji

git log --pretty=format:"%h - %an, %ar : %s" - pozwala skonfigurować własne parametry wyświetlania (tutaj skrócona suma kontrolna, nazwisko autora, względna data i temat)

git log --pretty=format:"%h" --graph - dodatkowa oplca --graph tworzy graficzne odwzorowanie historii pracy a gałęziach

git log --since=2.weeks - pozwala ograniczyć wyniki wyszukiwania w czasie (since, after, until, before, author, committer)

git reset HEAD file.name - powoduje usunięcie z poczekalni pliku, który został tam dodany za pomocą git add

git stash -

git tag - wyświetla wszystkie etykiety w porządku alfabetycznym

git tag -l 'v1.4.2.*' - filtr wyświtla wszystkie tagi z serii 1.4.2 (np. 1.4.2.1, 1.4.2.2 itp)

git tag -a v1.4 -m 'my version 1.4' - dodanie etykiery opisanej (flaga -a), posiada ona sumę kontrolną i nazwisko osoby etykietującej

git tag -s v1.5 -m 'my signed 1.5 tag' - flaga -s pozwala dodać klucz GNU Privacy Guard (GPG)

git show v1.5 - pokazuje szczegóły wybranej etykiety

git tag v1.4-lw - utworzenie etykiety lekkiej

git tag -v [nazwa-etykiety] - służy do weryfikacji poprawności etykiety, używa GPG

git tag -a v.1.2 9fceb02 - możemy dodać etykietę do historycznego commita, dołączamy do niego fragment sumy kontrolnej

git push origin [nazwa-etykiety] aby umieścić etykiety na serwerze, należy je wypchnąć osobno

git push origin --tags Wysłanie na serwer wszystkich etykiet na raz

git config --global alias.co checkout - komenda alias.skrót pozwala ustawić skrót dla wybranego polecenia, używanie skrótów i aliasów przyspiesza pracę

git branch - powoduje wylistowanie wszystkich dostępnych aktualnie gałęzi. Domyślnie jest to tylko master

git branch nazwa-nowego-brancha - np. git branch testNav, polecenie tworzy nową gałąź, na której zapisujemy wyniki pracy prowadzonej równolegle do mastera. Teraz gdy wpiszemy "git branch" zobaczymy dwie gałęzie: master i testNav

git checkout nazwa-brancha - np. git checkout testNav - przełączy nas na tę nową gałąź i pozwoli na niejzapisywać commity

git checkout -b nazwa-nowego-brancha - pozwala utworzyć nowy branch i za jednym razem od razu się na niego przełączyć

git merge nazwa-brancha - polecenie powoduje scalenie wybranego brancha z masterem czyli dogranie wytworzonych na tym branchu zmian do mastera, aby wykonać taki merge należy być na masterze

git branch -d nazwa-brancha - usuwamy branch nie jest już potrzebny a wszystkie wprowadzone na nim zmiany zostały odrzucone lub zatwierdzone i zmergowane do mastera

git branch -D nazwa-brancha - usuwamy branch, który posiada niezapisane zmiany. -D pozwoli usunąć go razem z niezapisanymi zmianami

git merge master - jeśli master został zmieniony z innej strony a pracujesz na innym branchu i chciałbyś pobrać do tego brancha najnowsze zmiany z mastera użyj "git merge master" (merguj z mastera do brancha na którym aktualnie jestem)

git branch -v - pokazuje listę branchy (gałęzi) z ostatnio zatwierdzanymi zmianami

git branch --merged - pokazuje listę gałęzi które zostały już scalone (merge) z masterem

git branch --no-merged - pokazuje listę gałęzi które jeszcze nie zostały scalone (merge) z masterem

git fetch - pobiera z serwera wszystkie dane, których jeszczenie ma lokalnie aktualizując tym samym lokalne repozytorium do najnowszej wersji

git push origin myBranch - wysyłanie zmian na zdalny branch

git push origin myBranch:myOtherBranch - wysyłanie zmian z brancha lokalnego x na branch zdalny y

git push origin :myBranch - usunięcie wybranego brancha lokalnego

git checkout -b myNewBranch origin/myNewBranch - tworzenie gałęzi lokalnej bazującej na gałęzi zdalnej

git checkout -b myNewB origin.myNewBranch - tworząc gałąź na podstawie innej, możemy ten nowej nadać własną nazwę

git checkout -b myNewBranch origin/myNewBranch - uruchamia śledzenie zdalnego brancha tworząc połączenie z jego lokalnym odpowiednikiem

git checkout --track origin/myNewBranch - wersja skrócona podpięcia śledzenia zdalnej gałęzi

git branch myNewBranch --track origin/myNewBranch - tworzenie nowego brancha z uruchomieniem śledzenia wybranego brancha zdalnego

git rebase master uruchamiamy będąc na wybranym branchu, git bierze wprowadzone na branchu zmiany, cofa się do ostatniego wspólnego commita i do niego wprowadza te zapisane zmiany
